# Step 4: Define hyperparameter space

from scipy.stats import randint
from sklearn.model_selection import train_test_split, RandomizedSearchCV
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import accuracy_score, precision_score, recall_score, f1_score, confusion_matrix, ConfusionMatrixDisplay



param_dist = {
    'n_estimators': randint(50, 200),
    'max_depth': randint(3, 20),
    'min_samples_split': randint(2, 10),
    'min_samples_leaf': randint(1, 10),
    'bootstrap': [True, False]
}

model = RandomForestClassifier(random_state=42)

random_search = RandomizedSearchCV(
    model, param_distributions=param_dist, 
    n_iter=20, cv=5, verbose=1, n_jobs=-1, random_state=42
)
# Train base model BEFORE tuning
model.fit(X_train, y_train)

# Evaluation BEFORE Tuning
y_pred_before = model.predict(X_test)
print("Evaluation BEFORE Tuning:")
print("Accuracy:", accuracy_score(y_test, y_pred_before))
print("Precision:", precision_score(y_test, y_pred_before, zero_division=1))
print("Recall:", recall_score(y_test, y_pred_before, zero_division=1))
print("F1 Score:", f1_score(y_test, y_pred_before, zero_division=1))

# Summary of Findings

#  What Preprocessing Helped?

# Encoding Categorical Features:Converting categorical variables like gender and smoking_history using one-hot encoding was essential. Without this, the model failed with errors like ValueError: could not convert string to float.
# Feature Scaling: Standardizing numeric features using StandardScaler() helped ensure all features contributed equally to the model, especially important for distance-based models or algorithms sensitive to scale.
# Handling Missing Values (if any): Although this dataset was clean, in general, checking for and handling missing values is crucial before model training.

# What Parameters Improved Model Performance?

# 1. Using RandomizedSearchCV, the following hyperparameters from the RandomForestClassifier improved model performance:
# 2. n_estimators: More trees improved stability and accuracy.
# 3. max_depth: Controlled overfitting by limiting how deep each tree can grow.
# 4. min_samples_split and min_samples_leaf: Helped reduce overfitting by setting a minimum number of samples required to split or be at a leaf node.
# 5. bootstrap: Helped by using subsets of data (bootstrapping), improving generalization.
